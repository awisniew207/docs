title: Policy Precheck Function
---

The precheck function provides fast, local validation before committing to expensive Lit Action execution. It's your Policy's first line of defense against invalid requests.

## Purpose and Execution

<Steps>
  <Step title="Called by Ability Precheck">
    Your Policy precheck runs during the Ability's precheck phase
  </Step>

  <Step title="Local Execution">
    Runs in the executor's environment, not in Lit Actions
  </Step>

  <Step title="Fast Validation">
    Should complete quickly to provide good user experience
  </Step>

  <Step title="Best-Effort Check">
    Provides reasonable confidence that evaluate function will succeed
  </Step>
</Steps>

## Function Structure

```typescript
precheck: async ({ abilityParams, userParams }, policyContext) => {
  try {
    // Parameter validation
    const validation = await validateParameters(abilityParams, userParams);

    if (validation.success) {
      return policyContext.allow(validation.data);
    } else {
      return policyContext.deny(validation.error);
    }
  } catch (error) {
    // Unhandled errors automatically become deny results
    return policyContext.deny({
      reason: 'Precheck validation failed',
      error: error.message,
    });
  }
}
```

## Common Validation Patterns

### Spending Limit Validation

```typescript
precheck: async ({ abilityParams, userParams }, policyContext) => {
  const { tokenAddress, amount, usdValue } = abilityParams;
  const { dailySpendingLimit, allowedTokens } = userParams;

  // Token allowlist check
  if (allowedTokens && !allowedTokens.includes(tokenAddress)) {
    return policyContext.deny({
      reason: 'Token not in allowlist',
      tokenAddress,
      allowedTokens,
    });
  }

  // Get current daily spending
  const currentSpending = await getCurrentDailySpending({
    userAddress: policyContext.delegation.delegatorPkpInfo.ethAddress,
    appId: policyContext.appId,
  });

  // Check if this transaction would exceed limit
  if (currentSpending + usdValue > dailySpendingLimit) {
    return policyContext.deny({
      reason: 'Would exceed daily spending limit',
      currentSpending,
      requestedAmount: usdValue,
      dailyLimit: dailySpendingLimit,
      remainingLimit: Math.max(0, dailySpendingLimit - currentSpending),
    });
  }

  return policyContext.allow({
    currentSpending,
    remainingLimit: dailySpendingLimit - currentSpending - usdValue,
    tokenValidated: true,
  });
}
```

### Time-Based Access Control

```typescript
precheck: async ({ abilityParams, userParams }, policyContext) => {
  const { timestamp } = abilityParams;
  const { allowedHours, timezone, weekdaysOnly } = userParams;

  const now = new Date(timestamp || Date.now());

  // Apply timezone if specified
  if (timezone) {
    // Convert to user's timezone for validation
    const userTime = new Date(now.toLocaleString("en-US", { timeZone: timezone }));
    now.setTime(userTime.getTime());
  }

  const currentHour = now.getHours();
  const currentDay = now.getDay(); // 0 = Sunday, 6 = Saturday

  // Check allowed hours
  if (allowedHours && !allowedHours.includes(currentHour)) {
    return policyContext.deny({
      reason: 'Outside allowed hours',
      currentHour,
      allowedHours,
      timezone: timezone || 'UTC',
    });
  }

  // Check weekday restriction
  if (weekdaysOnly && (currentDay === 0 || currentDay === 6)) {
    return policyContext.deny({
      reason: 'Weekends not allowed',
      currentDay: ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'][currentDay],
    });
  }

  return policyContext.allow({
    currentTime: now.toISOString(),
    timezone: timezone || 'UTC',
    accessGranted: true,
  });
}
```

### Rate Limiting Check

```typescript
precheck: async ({ abilityParams, userParams }, policyContext) => {
  const { operationType, timestamp } = abilityParams;
  const { maxPerHour, maxPerDay, cooldownMinutes } = userParams;

  const userAddress = policyContext.delegation.delegatorPkpInfo.ethAddress;
  const now = timestamp || Date.now();

  // Check recent executions
  const recentExecutions = await getRecentExecutions({
    userAddress,
    appId: policyContext.appId,
    operationType,
    since: now - (24 * 60 * 60 * 1000), // Last 24 hours
  });

  // Check hourly limit
  const lastHour = recentExecutions.filter(
    exec => exec.timestamp > now - (60 * 60 * 1000)
  );

  if (lastHour.length >= maxPerHour) {
    const oldestInHour = Math.min(...lastHour.map(e => e.timestamp));
    const resetTime = oldestInHour + (60 * 60 * 1000);

    return policyContext.deny({
      reason: 'Hourly rate limit exceeded',
      currentCount: lastHour.length,
      maxPerHour,
      resetTime: new Date(resetTime).toISOString(),
    });
  }

  // Check daily limit
  if (recentExecutions.length >= maxPerDay) {
    const oldestToday = Math.min(...recentExecutions.map(e => e.timestamp));
    const resetTime = oldestToday + (24 * 60 * 60 * 1000);

    return policyContext.deny({
      reason: 'Daily rate limit exceeded',
      currentCount: recentExecutions.length,
      maxPerDay,
      resetTime: new Date(resetTime).toISOString(),
    });
  }

  // Check cooldown period
  if (cooldownMinutes > 0 && recentExecutions.length > 0) {
    const lastExecution = Math.max(...recentExecutions.map(e => e.timestamp));
    const cooldownEnd = lastExecution + (cooldownMinutes * 60 * 1000);

    if (now < cooldownEnd) {
      return policyContext.deny({
        reason: 'Cooldown period active',
        lastExecution: new Date(lastExecution).toISOString(),
        cooldownEnd: new Date(cooldownEnd).toISOString(),
        remainingCooldown: Math.ceil((cooldownEnd - now) / 1000 / 60), // minutes
      });
    }
  }

  return policyContext.allow({
    hourlyCount: lastHour.length,
    dailyCount: recentExecutions.length,
    lastExecution: recentExecutions.length > 0
      ? new Date(Math.max(...recentExecutions.map(e => e.timestamp))).toISOString()
      : null,
  });
}
```

### Robust Error Handling Example

```typescript
precheck: async ({ abilityParams, userParams }, policyContext) => {
  const TIMEOUT_MS = 5000; // 5 second timeout
  const MAX_RETRIES = 3;

  for (let attempt = 1; attempt <= MAX_RETRIES; attempt++) {
    try {
      // Create timeout promise
      const timeoutPromise = new Promise((_, reject) => {
        setTimeout(() => reject(new Error('Validation timeout')), TIMEOUT_MS);
      });

      // Race validation against timeout
      const validationPromise = performValidation(abilityParams, userParams);
      const result = await Promise.race([validationPromise, timeoutPromise]);

      return policyContext.allow(result);

    } catch (error) {
      const isLastAttempt = attempt === MAX_RETRIES;
      const isTimeoutError = error.message === 'Validation timeout';
      const isRetryableError = error.code === 'NETWORK_ERROR' || isTimeoutError;

      if (isLastAttempt || !isRetryableError) {
        return policyContext.deny({
          reason: 'Validation failed after retries',
          error: error.message,
          attempts: attempt,
        });
      }

      // Wait before retry (exponential backoff)
      await new Promise(resolve => setTimeout(resolve, 1000 * Math.pow(2, attempt - 1)));
    }
  }
}
```

### Optimized Precheck Example

```typescript
precheck: async ({ abilityParams, userParams }, policyContext) => {
  // Run multiple validations in parallel
  const [
    tokenValidation,
    spendingValidation,
    rateValidation
  ] = await Promise.allSettled([
    validateTokenAllowlist(abilityParams.tokenAddress, userParams.allowedTokens),
    validateSpendingLimit(abilityParams.amount, userParams.dailyLimit, policyContext),
    validateRateLimit(abilityParams.operationType, userParams.rateLimit, policyContext)
  ]);

  // Check results and return first failure
  if (tokenValidation.status === 'rejected' || !tokenValidation.value.allowed) {
    return policyContext.deny(tokenValidation.value.error);
  }

  if (spendingValidation.status === 'rejected' || !spendingValidation.value.allowed) {
    return policyContext.deny(spendingValidation.value.error);
  }

  if (rateValidation.status === 'rejected' || !rateValidation.value.allowed) {
    return policyContext.deny(rateValidation.value.error);
  }

  // All validations passed
  return policyContext.allow({
    tokenValidated: tokenValidation.value.data,
    spendingValidated: spendingValidation.value.data,
    rateValidated: rateValidation.value.data,
  });
}
```
