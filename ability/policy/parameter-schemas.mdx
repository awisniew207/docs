## abilityParamsSchema

This schema defines what parameters Vincent Abilities must provide when your Policy evaluates their execution request.

```typescript
import { z } from 'zod';

const abilityParamsSchema = z.object({
  // Core transaction details
  tokenAddress: z.string().regex(/^0x[a-fA-F0-9]{40}$/, "Invalid token address"),
  amount: z.number().positive("Amount must be positive"),

  // Optional context
  recipientAddress: z.string().regex(/^0x[a-fA-F0-9]{40}$/).optional(),
  transactionType: z.enum(["transfer", "swap", "stake"]).optional(),
});
```

## userParamsSchema

This schema defines the user-configurable parameters that control your Policy's behavior. These parameters are:
- Set by users during delegation
- Stored securely on-chain
- Cannot be modified by Apps or executors

```typescript
const userParamsSchema = z.object({
  // Spending controls
  dailySpendingLimit: z.number().positive(),
  weeklySpendingLimit: z.number().positive().optional(),

  // Token restrictions
  allowedTokens: z.array(z.string()).optional(),
  blockedTokens: z.array(z.string()).optional(),

  // Time-based controls
  allowedHours: z.array(z.number().min(0).max(23)).optional(),
  timezone: z.string().optional(),
});
```

## Schema Integration

Your schemas work together to provide comprehensive validation:

```typescript
import { createVincentPolicy } from '@lit-protocol/vincent-ability-sdk';

const vincentPolicy = createVincentPolicy({
  packageName: '@example/spending-limit-policy',

  // Define what Abilities must provide
  abilityParamsSchema: z.object({
    tokenAddress: z.string(),
    amount: z.number().positive(),
    usdValue: z.number().positive(),
  }),

  // Define what users can configure
  userParamsSchema: z.object({
    dailySpendingLimit: z.number().positive(),
    allowedTokens: z.array(z.string()).optional(),
  }),

  // Use both in your policy functions
  precheck: async ({ abilityParams, userParams }, policyContext) => {
    const { tokenAddress, amount, usdValue } = abilityParams;
    const { dailySpendingLimit, allowedTokens } = userParams;

    // Validate against user configuration
    if (allowedTokens && !allowedTokens.includes(tokenAddress)) {
      return policyContext.deny({ reason: 'Token not allowed' });
    }

    // Check spending limits
    if (usdValue > dailySpendingLimit) {
      return policyContext.deny({ reason: 'Exceeds daily limit' });
    }

    return policyContext.allow({ validated: true });
  },
});
```

### Schema Evolution Example

```typescript
// Version 1 - Basic schema
const userParamsSchemaV1 = z.object({
  dailyLimit: z.number(),
});

// Version 2 - Extended with backward compatibility
const userParamsSchemaV2 = z.object({
  // Legacy support
  dailyLimit: z.number().optional(),

  // New structured approach
  limits: z.object({
    daily: z.number(),
    weekly: z.number().optional(),
    monthly: z.number().optional(),
  }).optional(),
}).refine((data) => {
  // Ensure either old or new format is provided
  return data.dailyLimit !== undefined || data.limits !== undefined;
}, {
  message: "Either dailyLimit or limits must be specified",
});
```

## Type Safety

Get TypeScript types from your schemas:

```typescript
import { z } from 'zod';

const abilityParamsSchema = z.object({
  tokenAddress: z.string(),
  amount: z.number(),
});

const userParamsSchema = z.object({
  dailyLimit: z.number(),
  allowedTokens: z.array(z.string()).optional(),
});

// Infer TypeScript types
type AbilityParams = z.infer<typeof abilityParamsSchema>;
// Result: { tokenAddress: string; amount: number; }

type UserParams = z.infer<typeof userParamsSchema>;
// Result: { dailyLimit: number; allowedTokens?: string[]; }

// Use in your policy functions
const precheck = async (
  { abilityParams, userParams }: {
    abilityParams: AbilityParams;
    userParams: UserParams
  },
  policyContext: PolicyContext
) => {
  // Full type safety and IntelliSense
  const isAllowed = userParams.allowedTokens?.includes(abilityParams.tokenAddress);
  // ...
};
```

## Testing Your Schemas

Validate your schemas work correctly:

```typescript
// Test ability parameter validation
const testAbilityParams = {
  tokenAddress: '0x1234567890123456789012345678901234567890',
  amount: 100,
};

const abilityResult = abilityParamsSchema.safeParse(testAbilityParams);
if (!abilityResult.success) {
  console.error('Ability params validation failed:', abilityResult.error.issues);
}

// Test user parameter validation
const testUserParams = {
  dailyLimit: 1000,
  allowedTokens: ['0x1234567890123456789012345678901234567890'],
};

const userResult = userParamsSchema.safeParse(testUserParams);
if (!userResult.success) {
  console.error('User params validation failed:', userResult.error.issues);
}
```
